int helper(int a, int b)
{
    if (a <= b)
        return 0;
    else
        return 1 + helper(a / 2, b);
}

int carpetBox(int a, int b, int c, int d)
{
    return min(helper(a, c) + helper(b, d) , helper(a, d) + helper(b, c));
}
